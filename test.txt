import os
import sys

def resource_path(relative_path):
    try:
        base_path = sys._MEIPASS  # PyInstaller unpacked temp path
    except AttributeError:
        base_path = os.path.abspath(".")
    return os.path.join(base_path, relative_path)

# Get the full path to your Streamlit script
streamlit_app = resource_path("myapp/streamlit_app.py")

# Launch Streamlit using os.system
os.system(f'"{sys.executable}" -m streamlit run "{streamlit_app}"')
