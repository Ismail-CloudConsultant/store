import streamlit as st

st.set_page_config(page_title="Four Sheets App", layout="wide")

st.title("Four Sheets App")

# Create four tabs
tabs = st.tabs(["Create", "Update", "Validate", "Visualize"])

# Tab 1 - Create
with tabs[0]:
    st.header("Create")
    st.write("This is the Create sheet.")
    # Example UI
    name = st.text_input("Enter a name")
    if st.button("Create Entry"):
        st.success(f"Created entry: {name}")

# Tab 2 - Update
with tabs[1]:
    st.header("Update")
    st.write("This is the Update sheet.")
    record_id = st.number_input("Enter record ID to update", min_value=1)
    new_value = st.text_input("New Value")
    if st.button("Update Entry"):
        st.info(f"Updated record {record_id} with value '{new_value}'.")

# Tab 3 - Validate
with tabs[2]:
    st.header("Validate")
    st.write("This is the Validate sheet.")
    uploaded_file = st.file_uploader("Upload file for validation", type=["csv", "xlsx"])
    if uploaded_file:
        st.success("File uploaded and ready for validation.")

# Tab 4 - Visualize
with tabs[3]:
    st.header("Visualize")
    st.write("This is the Visualize sheet.")
    st.line_chart({"Data": [1, 3, 2, 4, 5, 3, 4]})
