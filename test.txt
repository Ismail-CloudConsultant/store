import streamlit as st

# Page config
st.set_page_config(layout="wide")

# CSS for button styling to look like your diagram
st.markdown("""
    <style>
        .menu-container {
            display: flex;
            flex-direction: column;
            gap: 10px;
            padding: 10px;
            border-right: 2px solid black;
            height: 100vh;
        }
        .menu-button {
            display: block;
            width: 100%;
            padding: 10px;
            border: 2px solid black;
            background-color: white;
            text-align: center;
            font-size: 16px;
            cursor: pointer;
        }
        .menu-button:hover {
            background-color: #f0f0f0;
        }
    </style>
""", unsafe_allow_html=True)

# Create two columns: left menu and right content
col1, col2 = st.columns([1, 4])

with col1:
    st.markdown('<div class="menu-container">', unsafe_allow_html=True)

    if st.button("Create", key="create"):
        st.session_state.page = "Create"

    if st.button("Update", key="update"):
        st.session_state.page = "Update"

    if st.button("Validate", key="validate"):
        st.session_state.page = "Validate"

    if st.button("Visualization", key="visualization"):
        st.session_state.page = "Visualization"

    st.markdown('</div>', unsafe_allow_html=True)

with col2:
    st.write("")  # Blank space like in your image
