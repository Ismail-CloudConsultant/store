import streamlit as st

st.set_page_config(page_title="Four Sheets App", layout="wide")

st.sidebar.title("Navigation")

# Define a function to create box-styled radio buttons
def sidebar_box(label, options, key):
    box = st.sidebar.container()
    box.markdown(
        f"""
        <div style="
            padding: 10px;
            border: 1px solid #ccc;
            border-radius: 8px;
            margin-bottom: 10px;
            background-color: #f9f9f9;
        ">
        <strong>{label}</strong>
        </div>
        """,
        unsafe_allow_html=True
    )
    return st.sidebar.radio("", options, key=key)

# Each option as its own box
create_selected = sidebar_box("Create Sheet", ["Go to Create"], key="create")
update_selected = sidebar_box("Update Sheet", ["Go to Update"], key="update")
validate_selected = sidebar_box("Validate Sheet", ["Go to Validate"], key="validate")
visualize_selected = sidebar_box("Visualize Sheet", ["Go to Visualize"], key="visualize")

# Logic to switch pages
if create_selected == "Go to Create":
    sheet = "Create"
elif update_selected == "Go to Update":
    sheet = "Update"
elif validate_selected == "Go to Validate":
    sheet = "Validate"
elif visualize_selected == "Go to Visualize":
    sheet = "Visualize"
else:
    sheet = "Create"

# Main content
st.title("Four Sheets App")

if sheet == "Create":
    st.header("Create")
    name = st.text_input("Enter a name")
    if st.button("Create Entry"):
        st.success(f"Created entry: {name}")

elif sheet == "Update":
    st.header("Update")
    record_id = st.number_input("Enter record ID to update", min_value=1)
    new_value = st.text_input("New Value")
    if st.button("Update Entry"):
        st.info(f"Updated record {record_id} with value '{new_value}'.")

elif sheet == "Validate":
    st.header("Validate")
    uploaded_file = st.file_uploader("Upload file for validation", type=["csv", "xlsx"])
    if uploaded_file:
        st.success("File uploaded and ready for validation.")

elif sheet == "Visualize":
    st.header("Visualize")
    st.line_chart({"Data": [1, 3, 2, 4, 5, 3, 4]})
