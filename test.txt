import pandas as pd

# Sample data
df = pd.DataFrame({
    'A': ['123', '45a', '67!', '89', 'abc', '100'],
    'B': ['x', 'y', 'z', 'w', 'v', 'u']
})

# Keep only rows where A is a valid integer (optional: drop others)
df_cleaned = df[df['A'].str.match(r'^\d+$')].copy()

# Convert column A to integers
df_cleaned['A'] = df_cleaned['A'].astype(int)

print(df_cleaned)



df['A'] = pd.to_numeric(df['A'], errors='coerce')
