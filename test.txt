import pandas as pd
from datetime import datetime

# Sample data
df = pd.DataFrame({
    'date': ['12-05-2024', '05/12/2024', '01-01-2025', '07/04/2025']
})

# Function to parse and reformat
def convert_to_mmddyyyy(date_str):
    try:
        if '-' in date_str:
            dt = datetime.strptime(date_str, "%d-%m-%Y")
        elif '/' in date_str:
            dt = datetime.strptime(date_str, "%m/%d/%Y")
        else:
            return date_str  # Unrecognized format
        return dt.strftime("%m/%d/%Y")
    except Exception as e:
        return date_str  # Or return None / log error if preferred

# Apply the function
df['formatted_date'] = df['date'].apply(convert_to_mmddyyyy)

print(df)
