import sys
from PyQt6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QPushButton,
    QFileDialog, QLabel, QTextEdit
)

class FileUploader(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Upload Two Files")
        self.setGeometry(100, 100, 800, 600)

        self.layout = QVBoxLayout()

        # Buttons
        self.upload_button1 = QPushButton("Upload File 1")
        self.upload_button2 = QPushButton("Upload File 2")

        # File labels
        self.file1_label = QLabel("No file uploaded yet.")
        self.file2_label = QLabel("No file uploaded yet.")

        # Text displays
        self.file1_text = QTextEdit()
        self.file1_text.setReadOnly(True)

        self.file2_text = QTextEdit()
        self.file2_text.setReadOnly(True)

        # Add widgets to layout
        self.layout.addWidget(self.upload_button1)
        self.layout.addWidget(self.file1_label)
        self.layout.addWidget(self.file1_text)

        self.layout.addWidget(self.upload_button2)
        self.layout.addWidget(self.file2_label)
        self.layout.addWidget(self.file2_text)

        # Set layout
        self.setLayout(self.layout)

        # Connect buttons
        self.upload_button1.clicked.connect(lambda: self.upload_file(1))
        self.upload_button2.clicked.connect(lambda: self.upload_file(2))

    def upload_file(self, file_number):
        file_path, _ = QFileDialog.getOpenFileName(self, "Open File")

        if file_path:
            try:
                with open(file_path, "r", encoding="utf-8") as f:
                    content = f.read()
            except Exception as e:
                content = f"Error reading file: {e}"

            if file_number == 1:
                self.file1_label.setText(f"File 1: {file_path}")
                self.file1_text.setText(content)
            elif file_number == 2:
                self.file2_label.setText(f"File 2: {file_path}")
                self.file2_text.setText(content)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = FileUploader()
    window.show()
    sys.exit(app.exec())
